# Leggo due numeri
a = int(input("Inserisci il valore di a:"))
b = int(input("Inserisci il valore di b:"))

if a < b:
    for i in range(a, b+1):
        print(i)
else:
    for i in range(b, a+1, -1):
        print(i)
